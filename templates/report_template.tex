\renewcommand{\maketitle}{%
  \begin{titlepage}
    \newgeometry{margin=0cm}
    \begin{tikzpicture}[remember picture,overlay]
        \IfFileExists{$heroimage$}{%
            \node[at=(current page.center),inner sep=0]{\includegraphics[width=\paperwidth,height=\paperheight]{$heroimage$}};
        }{%
            % Optional: Add a fallback color or text if image is missing
        }
      \fill[black,opacity=0.6] (current page.south west) rectangle (current page.north east);
      \node[at=(current page.center),text=white,align=center]{
        \begin{minipage}{0.8\textwidth}
          \centering
          \IfFileExists{$logo$}{%
            \includegraphics[height=3cm]{$logo$}\par\vspace{2cm}
          }{%
            % Optional: Fallback for logo
          }
          {\Huge\bfseries\thetitle\par}\vspace{1cm}
          \vspace{1.5cm}
          {\color{LeliaOrange}\rule{0.2\textwidth}{2pt}\par}\vspace{0.5cm}
          {\large\thedate\par}
        \end{minipage}
      };
    \end{tikzpicture}
    \restoregeometry
  \end{titlepage}
}
